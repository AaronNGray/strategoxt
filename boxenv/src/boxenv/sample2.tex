\documentclass{article}
\usepackage{boxenv}

\def\linea{%
\begin{HBOX}{1}%
This~is~a~long~horizontal~line~of~text~which~may~exceed~the~right~margin~%
an%d~wit%h%n%~an~ALT~box~the~alternative~box~should~be~selected~by~\LaTeX{}.%
\end{HBOX}%
}

\long\def\lineb{%
\begin{HBOX}{1}%
This~is~a~long~horizontal~line~of~text~which~may~exceed~the~right~margin~%
a%nd~%w%i%t%h%n%~an~ALT~box~the~alternative~box~should~be~selected~by~\LaTeX{}.%
\end{HBOX}%
}

\begin{document}
\vspace{2cm}

\framebox[.5\linewidth]{%
\begin{minipage}{.5\linewidth}%
\ \\%
\begin{boxenv}[.5\linewidth]%
\begin{HBOX}{1}%{3}%
LM~\begin{VBOX}{3}{0}%
\linea%

\begin{VBOX}{0}{0}%
\begin{ALTBOX}%
{\linea}%
{Alternative text is selected}%
\end{ALTBOX}%

\begin{LBOX}{=}%
\begin{ALTBOX}%
{\linea}%
{Alternative text is selected}%
\end{ALTBOX}%
\end{LBOX}%
\end{VBOX}%

\begin{VBOX}{0}{0}%
\begin{ALTBOX}%
{\lineb}%
{Alternative text is selected}%
\end{ALTBOX}%

\begin{LBOX}{=}%
\begin{ALTBOX}%
{\lineb}%
{Alternative text is selected}%
\end{ALTBOX}%
\end{LBOX}%
\end{VBOX}%

\end{VBOX}%
\end{HBOX}%
\end{boxenv}%
\end{minipage}%
}
\end{document}
