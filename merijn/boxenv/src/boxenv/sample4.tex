\documentclass{article}

\usepackage{boxenv}

\def\line#1{%
\begin{HBOX}{0}%
line#1%
\end{HBOX}%
}

\def\linea{%
\begin{VBOX}{0}{0}%
\line{a}%

\begin{LBOX}{=}%
\line{a}%
\end{LBOX}%
\end{VBOX}%
}

\def\lineb{%
\begin{VBOX}{0}{0}%
\line{b}%

\begin{LBOX}{=}%
\line{b}%
\end{LBOX}%
\end{VBOX}%
}

\def\linec{%
\begin{VBOX}{0}{0}%
\begin{HBOX}{6}%
\linea~\lineb%
\end{HBOX}

\begin{LBOX}{=}%
\begin{HBOX}{6}%
\linea~\lineb%
\end{HBOX}%
\end{LBOX}%
\end{VBOX}
}

\def\lineda{%
\begin{VBOX}{0}{24}%
\linec%

linec%
\end{VBOX}%
}

\def\lined{%
\begin{VBOX}{0}{0}%
\lineda%

\begin{LBOX}{=}%
\lineda%
\end{LBOX}%
\end{VBOX}
}

\begin{document}

\noindent
This sample is an extension to sample1. It has nested width calculations
which are oriented horizontally.  The strings ``linea'' and ``lineb'' are
both underlined. Together both strings (and corresponding lines) are
oriented horizontally and underlined as well. The string ``linec'' is
displayed right to the other strings and a bar of total width underlines
everything.
\vspace{2cm}

\noindent
\begin{boxenv}
\begin{VBOX}{0}{1}
line0

\lined
\end{VBOX}

\end{boxenv}
\end{document}
